
# XCHG
xchg a, b intercambia el valor de a con el de b
xchg eax, esi : 

xchg eax, [esi] :
	pre:
		eax: 55
		esi: 0x100000 -> 12
	post:
		eax: 12
		esi: 55

xchg eax, ds:dword_4020DC
	pre:
		eax: 0x100000
		ds:dword_4020DC -> 12
	post:
		eax: 12
		ds:dword_4020DC -> 0x100000
